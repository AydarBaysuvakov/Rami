функция шаҡмаҡ(һан а)
	{
	ҡайтар а * а;
	};

функция дискриминант(һан a, һан b, һан c)
    {
    һан d ул шаҡмаҡ(b) - 4 * a * c;
    ҡайтар d ;
    };

яҙырға дискриминант(1, 2, 3);

рәт һандар[10];

һан һанау ул 0;
әле һанау < 10:
    {
    ҙурайт һанау;
    яҙырға һанау;
    һандар[һанау] ул һанау + 5;
    };
